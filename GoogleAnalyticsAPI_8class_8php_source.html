499  protected function generateSignedJWT() {
500 
501  // Check if a valid privateKey file is provided
502  if (!file_exists($this->privateKey) || !is_file($this->privateKey)) {
503  throw new Exception('Private key does not exist');
504  }
505 
506  // Create header, claim and signature
507  $header = array(
508  'alg' => 'RS256',
509  'typ' => 'JWT',
510  );
511 
512  $t = time();
513  $params = array(
514  'iss' => $this->email,
515  'scope' => GoogleOauth::SCOPE_URL,

518  'iat' => $t,
519  );

522  base64_encode(json_encode($header)),
523  base64_encode(json_encode($params)),
524  );
525 
526  // Compute Signature
527  $input = implode('.', $encodings);
528  $certs = array();

530  if (!openssl_pkcs12_read($pkcs12, $certs, $this->password)) {
531  throw new Exception('Could not parse .p12 file');
532  }
533  if (!isset($certs['pkey'])) {

535  }

537  if (!openssl_sign($input, $sig, $keyId, 'sha256')) {
538  throw new Exception('Could not sign data');
539  }
540 
541  // Generate JWT
